Add Tags to PDF

Example how to automatically add tags into the document.

using System;
using PDFixSDK.Pdfix;
namespace PDFix.App.Module
{
class AddTags
{
public static void Run(
String openPath, // source PDF document
String savePath, // output PDF document
String configPath, // configuration file
bool preflight // preflight document template
)
{
Pdfix pdfix = new Pdfix();
if (pdfix == null)
throw new Exception("Pdfix initialization fail");
PdfDoc doc = pdfix.OpenDoc(openPath, "");
if (doc == null)
throw new Exception(pdfix.GetError());
var doc_tmpl = doc.GetTemplate();
if (doc_tmpl == null)
throw new Exception(pdfix.GetError());
// load template configuration from json
PsFileStream stm = pdfix.CreateFileStream(configPath, PsFileMode.kPsReadOnly);
if (stm != null)
{
if (!doc_tmpl.LoadFromStream(stm, PsDataFormat.kDataFormatJson))
throw new Exception(pdfix.GetError());
stm.Destroy();
}
if (preflight)
{
// add some reference pages to preflight
for (var i = 0; i < doc.GetNumPages(); i++)
{
if (!doc_tmpl.AddPage(i, null, IntPtr.Zero))
throw new Exception(pdfix.GetError());
}
// run document template preflight
if (!doc_tmpl.Update(null, IntPtr.Zero))
throw new Exception(pdfix.GetError());
}
// define a cancel progress callback
PdfCancelProc cancel_callback = (data) =>
{
// to cancel the process return 1
Console.WriteLine("PdfCancelProc callback was called. The process can be cancelled here.");
return 0;
};
// define an event callback for changes in the page content
PdfEventProc event_callback = (data) =>
{
Console.WriteLine("Page contents did change.");
};
if (!pdfix.RegisterEvent(PdfEventType.kEventPageContentsDidChange, event_callback, IntPtr.Zero))
throw new Exception(pdfix.GetError());
if (!doc.RemoveTags(cancel_callback, IntPtr.Zero))
throw new Exception(pdfix.GetError());
if (!doc.AddTags(cancel_callback, IntPtr.Zero))
throw new Exception(pdfix.GetError());
if (!doc.Save(savePath, Pdfix.kSaveFull))
throw new Exception(pdfix.GetError());
if (!pdfix.UnregisterEvent(PdfEventType.kEventPageContentsDidChange, event_callback, IntPtr.Zero))
throw new Exception(pdfix.GetError());
doc.Close();
pdfix.Destroy();
}
}
}
